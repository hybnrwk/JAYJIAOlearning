\begin{equation*}
	\begin{aligned}
	\text{minimize}  & -x_1&+7x_2 &+3x_3 &\\
	\text{subject to}& x_1 &+3x_2  &-2x_3    &\leq 0 \\
	&x_1 &+4x_2  &-2x_3  &\geq 0  \\
	&-x_1 &+2x_2  &+3x_3    &=  5  \\
	&x_1 \leq 0, &x_2 \geq 0, &x_3  \text{无约束}&  
	\end{aligned}
\end{equation*}
